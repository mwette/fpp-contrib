\pgfdeclareshape{fingerrt}{%
  \anchor{center}{\pgfpointorigin}%
  \anchor{base}{\pgfpointorigin}%
  \anchor{tip}{\pgfpointorigin\advance\pgf@x+6pt}%
  \foregroundpath{%
    \pgfpathmoveto{\pgfpoint{+0pt}{+0pt}}%
    \pgfpathlineto{\pgfpoint{+0pt}{+6pt}}%
    \pgfpathlineto{\pgfpoint{+4pt}{+6pt}}%
    \pgfpathlineto{\pgfpoint{+4pt}{+4pt}}%
    \pgfpathlineto{\pgfpoint{+6pt}{+3pt}}%
    \pgfpathlineto{\pgfpoint{+6pt}{-3pt}}%
    \pgfpathlineto{\pgfpoint{+4pt}{-4pt}}%
    \pgfpathlineto{\pgfpoint{+4pt}{-6pt}}%
    \pgfpathlineto{\pgfpoint{+0pt}{-6pt}}%
    \pgfpathclose%
  }%
}
